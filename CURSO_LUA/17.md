# Exemplo 1: Classe simples


## Exemplo 2: Métodos

```lua
local function class(base)
    local c = {}
    c.__index = c
    setmetatable(c, {__index = base})

    function c:new(o)
        o = o or {}
        setmetatable(o, c)
        return o
    end
  return c
end

local Pessoa = class()
function Pessoa:saudar()
    print("Olá, meu nome é " .. self.nome)
end

local Aluno = class(Pessoa)
function Aluno:saudar()
  Pessoa.saudar(self)
  print("Sou aluno " .. self.nome .. " e estudo " .. self.curso)
end

local p = Pessoa:new({nome="Ana"})
local a = Aluno:new({nome="Bruno", curso="Lua"})
p:saudar()
a:saudar()
-- Olá, meu nome é Ana
-- Olá, meu nome é Bruno
-- Sou aluno Bruno e estudo Lua
```
